38 void draw_line(const DisplayVector display_a, const DisplayVector display_b, const Color color_a, const Color color_b, const Window &window = full_screen, size_t segments = 1) const
39 {
40 const Color color_ab = color_a * 0.5 + color_b * 0.5;
41
42 if (segments == 1)
43 {
44 /* project window display vector to full screen display vector to full screen pixel vector */
45 auto pixel_a = display_to_pixel_vector(window.display_to_display_vector(display_a));
46 auto pixel_b = display_to_pixel_vector(window.display_to_display_vector(display_b));
47 auto pixel_color = color_ab.to_R5G6B5();
48
49 /* call base driver */
50 _driver->drawLine(pixel_a.x,
51 pixel_a.y,
52 pixel_b.x,

55 }
56 else if (segments > 1)
57 {
58 const DisplayVector display_ab = display_a * 0.5 + display_b * 0.5;
59 draw_line(display_a, display_ab, color_a, color_ab, window, segments / 2);

99 void draw_polygon(DisplayVector display_center, DisplayVector display_radius, size_t sides, float rotation, Color color, const Window &window = full_screen, size_t segments = 1) const
100 {
101 for (size_t idx = 0; idx < sides; ++idx)

103 double start_angle_rad = ((static_cast<double>(idx) / static_cast<double>(sides)) - rotation) * 2 * M_PI;

225 DisplayVector pixel_to_display_vector(PixelVector pixel_vector) const
226 {
227 return {
228 static_cast<float>(static_cast<float>(pixel_vector.x) / _driver->width()),
229 static_cast<float>(static_cast<float>(pixel_vector.y) / _driver->height())
230 };
231 }

233 /* from normalized display vector to absolute pixel vector */
234 PixelVector display_to_pixel_vector(DisplayVector display_vector) const
235 {
236 PixelVector res{ static_cast<size_t>(display_vector.x * (_driver->width() - 1)),
237 static_cast<size_t>((1 - display_vector.y) * (_driver->height() - 1)) };
